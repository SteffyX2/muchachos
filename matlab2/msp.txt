bit = rand_bit(1,20);
 
A = [0 5];
T = 0.1;
deltaT = 0.01;
n = 0.25;
a = 0.5;
[s, t] = gen_impulsa(bit, A, T, deltaT);
 
figure
subplot(3,1,1)
plot(t, s)
axis tight
ylim([-1 A(2) + 1])
title('Pocetni signal')
 
sk = s;
for i = 1 : 15
	sk = kanal(sk, a, n);
	subplot(3,1,2)
	plot(t, sk)
	axis tight
	ylim([-1 A(2) + 1])
	title('Signal na ulazu')
 
	sk = pojacavac(sk, 1/a);
	subplot(3,1,3)
	plot(t, sk)
	axis tight
	ylim([-1 A(2) + 1])
	title('Signal na izlazu')
 
	pause
end
