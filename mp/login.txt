#include <LiquidCrystal.h>
LiquidCrystal lcd(8, 9, 4, 5, 6, 7);
unsigned char taster_staro, taster_novo;
char username[6] = "admin";
char password[6] = "pword";

void setup() {
  Serial.begin(9600);
  lcd.begin(16,2);
}
void loop() {
  char un[32], pw[32];
  int duzina;
	taster_novo = ocitaj_taster();
  if(taster_novo != taster_staro)
  {
	switch(taster_novo)
    {
      case 1: // SELECT
          break;
      case 2: // LEFT
          break;
      case 3: // UP
          break;
      case 4: // DOWN
          break;
      case 5: // RIGHT
          break;
      default:
          break;
    }
    
    taster_staro = taster_novo;
  }    
  Serial.print("Username: ");
  while(Serial.available()==0);
  delay(500);
  duzina = Serial.available();
  Serial.readBytes(un, duzina);
  un[duzina]=0;
  Serial.println(un);
  Serial.print("Password: ");
  while(Serial.available()==0);
  delay(500);
  duzina = Serial.available();
  Serial.readBytes(pw, duzina);
  pw[duzina]=0;
  Serial.println(pw);
  if (strcmp(un, username)==0 && strcmp(pw, password)==0)
  	lcd.print("Login successful");
  
}

byte ocitaj_taster()
{
  int tmp = analogRead(0); //stanje tastera se ocitava preko
  //analognog ulaza 0
  if (tmp > 635 && tmp < 645) //SELECT
    return 1;
  if (tmp > 405 && tmp < 415) //LEFT
    return 2;
  if (tmp > 95 && tmp < 105) //UP
    return 3;
  if (tmp > 252 && tmp < 262) //DOWN
    return 4;
  if (tmp < 5) //RIGHT
    return 5;
  return 0; //nije pritisnut nijedan od tastera
}
