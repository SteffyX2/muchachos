PROGRAM PLC_PRG
VAR_INPUT
	START AT %IX0.0 : BOOL;
	A1 AT %IX0.1 : BOOL;
	A2 AT %IX0.2 : BOOL;
END_VAR
VAR_OUTPUT
 	Y1 AT %QX0.0 : BOOL:=FALSE;
	H1 AT %QX0.1 : BOOL:=FALSE;
END_VAR
VAR
	BROJAC : CTU;
	TAJMER : TON;
	TRAJNI : BOOL:=FALSE;
END_VAR



BROJAC(CU:=A2, RESET:=START, PV:=6);
TAJMER(IN:=BROJAC.Q, PT:=T#5S);

IF START AND TAJMER.Q AND NOT BROJAC.Q THEN
	TRAJNI:=TRUE;
	H1:=FALSE;
END_IF

IF BROJAC.Q THEN
	TRAJNI:=FALSE;
	H1:=TRUE;
END_IF

IF TRAJNI AND A1 AND TAJMER.Q THEN
	Y1:=TRUE;
END_IF

IF A2 THEN
	Y1:=FALSE;
END_IF
